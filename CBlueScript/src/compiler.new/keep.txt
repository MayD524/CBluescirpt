char* pch; // pointer to the current token
pch = strtok(const_cast<char*>(line.c_str()), " (;");
do {
    token tk;
    //tk.isCmd = false;
    auto iter = BS::BS_BINDS.find(pch);
    if (iter != BS::BS_BINDS.end()) {
        tk.ivalue = iter->second;
        tk.isCmd = true;
    }
    else {
        tk.svalue = pch;
        tk.isCmd = false;
    }
    tk.svalue = pch;
    tokens.push_back(tk);
    pch = strtok(NULL, " (;");
} while (pch != NULL);
